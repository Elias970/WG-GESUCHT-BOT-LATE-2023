    /**
     * Called after a successfull request to the WebService, turns the results into
     * the appropriate object to display all the city names categorized by countryname
     *
     * @param obj response The WebService Response
     *
     * @return obj         City list as object
     */
    function prepareReceivedData(response) {
        response = $.makeArray(JSON.parse(response));
        return {
            suggestions: $.map(response, function(item, counter) {
                return { value: response[counter].city_and_state, data: {
                        'category'    : country_names[response[counter].country_code],
                        'city_id'     : response[counter].city_id,
                        'city_name'   : response[counter].city_name,
                        'country_code': response[counter].country_code
                    }
                };
            })
        };
    }


    /**
     * Displays a no_cities_found error message

    /**
     * Called when the city is changed but a page refresh does not happen. Replace the currency symbol based
     * on the country the city is in.
     *
     * PLEASE NOTE that this function is based on select_currency_based_on_country function found in
     * functions.inc.php, so if something is changed here, it should be changed there as well.
     *
     * @param str country_code The country code
     */
    function changeCurrencyBasedOnCountry(country_code) {
        switch(country_code) {
            case 'ch':
                currency = "CHF";
                break;
            default:
                currency = "\u20AC";
                break;
        }

        $('.currency_symbol').html(currency);

    /**
     * Called when on wohnungsmarkt/wohnungssuche etc where after selecting a new city a redirect is expected
     *
     * @param str city_name The city name
     * @param str city_id   The city id
     * @param str ad_type   The ad type can be either 'offers' or 'requests'
     */
    function redirectUserToListView(city_name, city_id, ad_type) {
        var categories = $('#categories.selectpicker').val();
        if (typeof categories === 'undefined') {
            categories = getAllCategories();
            $('#categories.selectpicker').val(categories);
        }
        if (categories.length === 0) {
            $('#no_categories_selected').fadeIn(400).delay(2000).fadeOut(400);
            return;
        }

        var url = 'https://www.wg-gesucht.de/';
        var language = 'de';

        var category_url_part = '';

        if (language !== 'de') {
            url += language + '/';
        }

        for (var i = 0; i < categories.length; i++) {
           if (categories[i] == 0){
               url += 'wg-zimmer';
           } else if (categories[i] == 1){
               url += '1-zimmer-wohnungen';
           } else if (categories[i] == 2){
               url += 'wohnungen';
           } else if (categories[i] == 3){
               url += 'haeuser';
           }

           category_url_part += categories[i];
           if (typeof categories[i + 1] != "undefined") {
               url += '-und-';
               category_url_part += '+';
           }
        }

        url += '-in-' + city_name;

        // build the url according to the ad_type
        if (ad_type === 'requests') {
            url += '-gesucht';
        }

        url += '.' + city_id + '.' + category_url_part + '.1.0.html';
        window.location.href = url;
    }


    /**
     * Makes a request to the Location WebService to get a list of districts for a city id
     *
     * @param obj services The services object
     * @param int city_id  The City id
     */
    function fetchDistrictsForCityId(services, city_id) {
        $.when(
            services.location.getDistrictsByCityId(city_id)
        ).done(function(response) {
            var districts = response._embedded.districts;

            if (districts.length == 0) {
                emptyFilterDistricts();
            } else {
                updateFilterDistricts(districts);
            }
        }).fail(function() {
            emptyFilterDistricts();
        });
    }


    /**
     * Repopulates the content of the filter_districts div with data retrieved from
     * the districts webService
     *
     * @param obj districts The districts
     */
    function updateFilterDistricts(districts) {
        var options = "";

        districts.forEach(function(district) {
            options += '<option value=' + district.district_id + ' style="padding: 4px 15px;">' + district.district_name + '</option>'
        });

        $("select[name^=ot]").html(options);
        $("select[name^=ot]").selectpicker('refresh');
    }


    /**
     * Empties the district filter selectbox
     */
    function emptyFilterDistricts() {
        $("select[name^=ot]").empty();
        $("select[name^=ot]").selectpicker('refresh');

        $('<div/>', {
            'style': 'padding: 10px 0;font-weight: bold;',
            'text': translation_lang_filters_districts.fdl_no_suburbs
        }).appendTo($('#filter_districts_list'));
    }


    /**
     * Updates the stadt_key and city_id hidden fields and performs an action
     * based on the city_select_box_action variable
     *
     * @param obj services   The services object
     * @param str suggestion The City name
     */
    function updateSelectBoxValue(suggestion) {
        $('#autocompinp').trigger('blur');

        $('#country_code').val(suggestion.data.country_code);
        $('#countrymanuel').val(suggestion.data.category);

        $('#label_city_name').text(suggestion.data.city_name);

        // remove umlauts from city name, breaks the url
        // Please check the functions.inc.php for a complete list of special characters
        // that need replacement and further instructions
        var city_name = replaceSpecialCharacters(suggestion.data.city_name);
        $('#city_name').val(city_name);

        var city_id = (suggestion.data.city_id);
        $('#stadt_key, #city_id').val(city_id).trigger('change');

        var city_select_box_action = 'form_submit';

        if (city_select_box_action !== '') {
            switch (city_select_box_action) {
                case 'form_submit':
                    disableHiddenInputsAndSubmitForm();
                    break;

                case 'redirect_requests_market':
                    redirectUserToListView(city_name, city_id, 'requests');
                    break;

                case 'redirect_offers_market':
                    redirectUserToListView(city_name, city_id, 'offers');
                    break;

                case 'user_filters_offer':
                    fetchDistrictsForCityId(services, city_id);
                    changeCurrencyBasedOnCountry(suggestion.data.country_code);
                    break;

                case 'user_filters_request':
                    fetchDistrictsForCityId(services, city_id);
                    changeCurrencyBasedOnCountry(suggestion.data.country_code);
                    break;

                case 'create_edit_ad':
                    changeCurrencyBasedOnCountry(suggestion.data.country_code);
                    break;
            }
        }
    }

    }
